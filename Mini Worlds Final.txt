    // Interval references for bubble generation on home and detail views
    let homeBubbleInterval;
    let detailBubbleInterval;

    /**
     * Creates a single floating bubble element with random properties
     * @param {string} container - ID of container to append bubble to
     */
    function createBubble(container = 'bubbleContainer') {
      const bubble = document.createElement('div');
      bubble.className = 'bubble';
      
      // Random size between 10px and 50px for visual variety
      const size = Math.random() * 40 + 10;
      bubble.style.width = size + 'px';
      bubble.style.height = size + 'px';
      
      // Random horizontal starting position
      bubble.style.left = Math.random() * 100 + '%';
      
      // Random animation duration between 8-15 seconds
      bubble.style.animationDuration = (Math.random() * 7 + 8) + 's';
      
      const targetContainer = document.getElementById(container);
      if (targetContainer) {
        targetContainer.appendChild(bubble);
        
        // Remove bubble from DOM after animation completes to prevent memory leaks
        setTimeout(() => {
          if (bubble.parentNode) {
            bubble.parentNode.removeChild(bubble);
          }
        }, 15000);
      }
    }

    /**
     * Starts bubble generation for the home screen
     */
    function startBubbles() {
      // Create initial batch of bubbles for immediate visual effect
      for (let i = 0; i < 30; i++) {
        setTimeout(() => createBubble('bubbleContainer'), i * 150);
      }
      
      // Continue creating bubbles at intervals while on home screen
      homeBubbleInterval = setInterval(() => {
        createBubble('bubbleContainer');
      }, Math.random() * 600 + 400);
    }

    /**
     * Starts bubble generation for the detail view
     */
    function startDetailBubbles() {
      // Clear any existing detail bubbles from previous detail view
      const detailContainer = document.getElementById('detailBubbleContainer');
      if (detailContainer) {
        detailContainer.innerHTML = '';
      }
      
      // Create initial batch of bubbles for detail view
      for (let i = 0; i < 25; i++) {
        setTimeout(() => createBubble('detailBubbleContainer'), i * 120);
      }
      
      // Continue creating bubbles for detail view
      detailBubbleInterval = setInterval(() => {
        createBubble('detailBubbleContainer');
      }, Math.random() * 700 + 500);
    }

    /**
     * Stops bubble generation for the detail view and clears existing bubbles
     */
    function stopDetailBubbles() {
      if (detailBubbleInterval) {
        clearInterval(detailBubbleInterval);
        detailBubbleInterval = null;
      }
      
      // Clear bubbles from detail container
      const detailContainer = document.getElementById('detailBubbleContainer');
      if (detailContainer) {
        setTimeout(() => {
          detailContainer.innerHTML = '';
        }, 500);
      }
    }

    /**
     * Opens the fullscreen image modal with the specified image and caption
     * @param {string} imageSrc - URL of the image to display
     * @param {string} caption - Caption text for the image
     */
    function openFullscreen(imageSrc, caption) {
      const modal = document.getElementById('fullscreenModal');
      const image = document.getElementById('fullscreenImage');
      const captionElement = document.getElementById('fullscreenCaption');
      
      // Set image source
      image.src = imageSrc;
      image.alt = caption;
      captionElement.textContent = caption;
      
      // Reset all image styles to ensure proper display
      image.style.cssText = '';
      image.className = 'max-w-full max-h-full object-contain rounded-lg shadow-2xl';
      
      // Show modal with fade-in animation
      modal.classList.remove('hidden');
      modal.classList.add('fullscreen-fade-in');
      
      // Prevent body scroll while modal is open
      document.body.style.overflow = 'hidden';
    }

    /**
     * Closes the fullscreen image modal
     */
    function closeFullscreen() {
      const modal = document.getElementById('fullscreenModal');
      
      // Add fade-out animation
      modal.classList.remove('fullscreen-fade-in');
      modal.classList.add('fullscreen-fade-out');
      
      // Wait for animation to complete before hiding
      setTimeout(() => {
        modal.classList.add('hidden');
        modal.classList.remove('fullscreen-fade-out');
        document.body.style.overflow = 'auto';
      });
    }

    // ============================================================
    // EVENT LISTENERS & APP STARTUP
    // ============================================================

    /**
     * Initializes fullscreen modal event listeners
     */
    function initFullscreen() {
      const closeButton = document.getElementById('closeFullscreen');
      const modal = document.getElementById('fullscreenModal');
      
      // Close button click handler
      closeButton.addEventListener('click', closeFullscreen);
      
      // Close on background click
      modal.addEventListener('click', (e) => {
        if (e.target === modal) {
          closeFullscreen();
        }
      });
      
      // Close on escape key
      document.addEventListener('keydown', (e) => {
        if (e.key === 'Escape' && !modal.classList.contains('hidden')) {
          closeFullscreen();
        }
      });
    }

      initApp();
      startBubbles();
      initFullscreen();

      document.addEventListener('DOMContentLoaded', initialize);
